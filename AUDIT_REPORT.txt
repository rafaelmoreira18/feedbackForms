[ARCH-02] Enforce role-based access control (RBAC)
  Current: User entity has a `role` field ('admin' | 'viewer') but it is never
           checked. Any valid JWT holder can access all protected routes equally.
  Better: Add a Roles guard and @Roles() decorator.
    // roles.guard.ts
    @Injectable()
    export class RolesGuard implements CanActivate {
      canActivate(context: ExecutionContext): boolean {
        const requiredRoles = this.reflector.get<string[]>('roles', context.getHandler());
        const { user } = context.switchToHttp().getRequest();
        return requiredRoles.includes(user.role);
      }
    }
    // In controller:
    @Delete()
    @Roles('admin')
    @UseGuards(JwtAuthGuard, RolesGuard)
    deleteAll() { ... }
  Impact: Prevents viewers from deleting data. Enforces least-privilege principle.
